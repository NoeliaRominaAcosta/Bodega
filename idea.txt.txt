// 🏪 BODEGA
class Bodega {
  constructor(nombre) {
    this.nombre = nombre;
    this.stock = [];
  }

  agregarVino(vino, cantidad) {
    const existente = this.stock.find(item => item.vino.id === vino.id);

    if (existente) {
      existente.cantidad += cantidad;
    } else {
      this.stock.push(new ItemStock(vino, cantidad));
    }
  }

  buscarVino(id) {
    return this.stock.find(item => item.vino.id === id);
  }

  hayStock(id, cantidad) {
    const item = this.buscarVino(id);
    return item && item.cantidad >= cantidad;
  }

  descontarStock(id, cantidad) {
    const item = this.buscarVino(id);
    if (item) {
      item.cantidad -= cantidad;
    }
  }

  mostrarStock() {
    console.log("📦 Stock actual:");
    this.stock.forEach(item => {
      console.log(`${item.vino.nombre} - ${item.cantidad} unidades`);
    });
  }
}

class Pedido {
  constructor(cliente, bodega) {
    this.cliente = cliente;
    this.bodega = bodega;
    this.items = [];
  }

  agregarProducto(idVino, cantidad) {
    if (!this.bodega.hayStock(idVino, cantidad)) {
      console.log("❌ No hay stock suficiente");
      return;
    }

    const itemStock = this.bodega.buscarVino(idVino);

    this.items.push({
      vino: itemStock.vino,
      cantidad: cantidad
    });

    this.bodega.descontarStock(idVino, cantidad);

    console.log(`✅ Agregado al pedido: ${itemStock.vino.nombre}`);
  }

  calcularTotal() {
    return this.items.reduce((total, item) => {
      return total + item.vino.precio * item.cantidad;
    }, 0);
  }

  mostrarPedido() {
    console.log(`🧾 Pedido de ${this.cliente.nombre}`);
    this.items.forEach(item => {
      console.log(`${item.vino.nombre} x${item.cantidad}`);
    });
    console.log(`💰 Total: $${this.calcularTotal()}`);
  }
}
